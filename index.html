1.1.7 Merge Waiuku

# Bind data from Waiuku site
waiuku <- bind_rows(
  nox_waiuku |> mutate(pollutant = "NOx", values = NOx),
  no2_waiuku |> mutate(pollutant = "NO2", values = NO2),
  no_waiuku |> mutate(pollutant = "NO", values = NO),
  pm10_waiuku |> mutate(pollutant = "PM10", values = PM10),
  pm2.5_waiuku |> mutate(pollutant = "PM2.5", values = PM2.5),
  airtemp_waiuku |> mutate(climate = "AirTemp", values = airtemp)
) |>
  select(date, pollutant, climate, values)
